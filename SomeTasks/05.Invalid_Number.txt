namespace _05.Invalid_Number
{
    class Program
    {
        static void Main(string[] args)
        {
            var number = double.Parse(Console.ReadLine());
            var isValid = (number >= 100 && number <= 200) || number == 0;     //true = !false  //  //false = !true

            if (!isValid)
            {
                Console.WriteLine("Pesho");
            }
        }
    }
}
